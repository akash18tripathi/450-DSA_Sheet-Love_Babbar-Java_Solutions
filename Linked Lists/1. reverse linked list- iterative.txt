public ListNode reverseList(ListNode head) {
    if(head==null){
        return null;
    }
    ListNode prev,curr,next;
    prev = null;
    curr = head;
    
    while(curr!=null){
        next = curr.next;
        curr.next = prev;
        prev = curr;
        curr = next;
    }
    head = prev;
    return head;
}
