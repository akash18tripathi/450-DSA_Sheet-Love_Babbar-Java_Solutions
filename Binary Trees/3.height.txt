int height(Node node) 
    {
        // code here 
        if(node==null){
            return 0;
        }
        if(node.left==null && node.right==null){
            return 1;
        }
        int leftHeight = height(node.left);
        int rightHeight = height(node.right);
        
        return Math.max(leftHeight,rightHeight)+1;
    }
